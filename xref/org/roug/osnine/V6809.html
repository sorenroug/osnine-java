75      /**
76       * Load modules.
77       * If the string starts with @ then it sets the address.
78       * If the string starts with $ then it is a long string of hex values.
79       * If the string starts with (srec) or (intel) then it expects the file
80       * to be Motorola S-record or Intel Hex format.
81       */
82      private static int loadModules(String[] files) throws Exception {
83          int loadAddress = 0;
84  
85          for (String fileToLoad : files) {
86              if (fileToLoad.startsWith("@")) {
87                  loadAddress = getVectorAddress(fileToLoad.substring(1));
88                  continue;
89              }
90              if (fileToLoad.startsWith("$")) {
91                  loadAddress = loadHexString(loadAddress, fileToLoad.substring(1));
92                  continue;
93              }
94              if (fileToLoad.toLowerCase().startsWith("(srec)")) {
95                  InputStream moduleStream = openFile(fileToLoad.substring(6));
96                  loadAddress = Loader.loadSRecord(moduleStream, bus);
97                  moduleStream.close();
98                  continue;
99              }
100             if (fileToLoad.toLowerCase().startsWith("(intel)")) {
101                 InputStream moduleStream = openFile(fileToLoad.substring(7));
102                 loadAddress = Loader.loadIntelHex(moduleStream, bus);
103                 moduleStream.close();
104                 continue;
105             }
106             //TODO Put the following into a method
107             LOGGER.debug("Loading {} at {}", fileToLoad, Integer.toHexString(loadAddress));
108 
109             InputStream moduleStream = openFile(fileToLoad);
110             int b = moduleStream.read();
111             while (b != -1) {
112                 bus.forceWrite(loadAddress, b);
113                 loadAddress++;
114                 b = moduleStream.read();
115             }
116             moduleStream.close();
117         }
118         LOGGER.debug("End address: {}", Integer.toHexString(loadAddress));
119         return loadAddress;
120     }

176     /**
177      * Installs a device. Looks for the Java class to load and the address
178      * in RAM to install it at.
179      * Also looks for additional arguments to provide to the constructor method
180      * of the device. Finally looks for additional attributes that can be set
181      * on the object.
182      * @param props The properties hashmap
183      * @param device The name of the device
184      */
185     private static void loadOneDevice(Properties props, String device)
186                                 throws Exception {
187         String deviceClsStr = props.getProperty(device + ".class");
188         int addr = getIntProperty(props, device + ".addr");
189         String deviceArgString = props.getProperty(device + ".args");
190         String[] argsList = new String[0];
191         if (deviceArgString != null) {
192             argsList = deviceArgString.split("\\s+");
193         }
194         LOGGER.debug("Loading {} class {}", device, deviceClsStr);
195         Class newClass = Class.forName(deviceClsStr);
196         Constructor<MemorySegment> constructor = newClass.getConstructor(
197                         Integer.TYPE, Bus8Motorola.class, String[].class);
198         MemorySegment deviceInstance = constructor.newInstance(addr, bus, argsList);
199         bus.insertMemorySegment(deviceInstance);
200         // Find additional setters.
201         String prefix = device + ".";
202         for (String key : props.stringPropertyNames()) {
203             if (key.startsWith(prefix)) {
204                 if (key.equals(prefix + "class") || key.equals(prefix + "addr")
205                         || key.equals(prefix + "args")) {
206                     continue;
207                 }
208                 String argument = props.getProperty(key);
209                 String methodName = generateMethodName(key.substring(prefix.length()));
210                 Method method = newClass.getMethod(methodName, String.class);
211                 method.invoke(deviceInstance, argument);
212             }
213         }
214     }

226     /**
227      * Load a hexstring into memory.
228      *
229      * @return the new location of the load pointer.
230      */
231     private static int loadHexString(int loadAddress, String input) {
232 	int len = input.length();
233 
234 	if (len == 0) {
235 	    return loadAddress;
236 	}
237 
238 	int startOffset;
239 	if (len % 2 != 0) {
240             bus.forceWrite(loadAddress, Character.digit(input.charAt(0), 16));
241 	    startOffset = 1;
242 	} else {
243 	    startOffset = 0;
244 	}
245 
246 	for (int i = startOffset; i < len; i += 2) {
247 	    bus.forceWrite(loadAddress + (i + 1) / 2,
248                     (Character.digit(input.charAt(i), 16) << 4)
249 		    + Character.digit(input.charAt(i + 1), 16));
250 	}
251         int loadSize = startOffset + (len / 2);
252         LOGGER.debug("Loading {} bytes at {}", loadSize, Integer.toHexString(loadAddress));
253         loadAddress += loadSize;
254         return loadAddress;
255     }
