36  /**
37   * Cause a periodic interrupt to the CPU. The IRQ is turned off when
38   * the CPU reads a specific memory address. (Typically $FF00)
39   */
40  public class IRQBeat extends MemorySegment {
41  
42  
43      private static final int CLOCKDELAY = 500;  // milliseconds
44      /** On Dragon 32 the interrupt is 50 times a second. */
45      private static final int CLOCKPERIOD = 20;  // milliseconds
46  
47      private static final Logger LOGGER = LoggerFactory.getLogger(IRQBeat.class);
48  
49      /** Reference to CPU for the purpose of sending IRQ. */
50      private Bus8Motorola bus;
51  
52      private ClockTick clocktask;
53  
54      /**
55       * Constructor.
56       *
57       * @param start - First address location of the ACIA.
58       * @param bus - The bus the ACIA is attached to.
59       * @param args - additional arguments
60       */
61      public IRQBeat(int start, Bus8Motorola bus, String... args) {
62          super(start, start + 1);
63          LOGGER.debug("Initialise heartbeat");
64          this.bus = bus;
65      }
66  
67      // Return 1 if IRQ is enabled and turn it off.
68      @Override
69      protected int load(int addr) {
70          LOGGER.debug("Check");
71          if (clocktask != null && clocktask.isActiveIRQ()) {
72              clocktask.setActiveIRQ(false);
73              bus.signalIRQ(false);
74              return 1; // There was an IRQ
75          } else {
76              return 0;
77          }
78      }
79  
80      @Override
81      protected void store(int addr, int val) {
82          LOGGER.debug("Starting heartbeat every {} milliseconds", val);
83          if (clocktask == null) {
84              clocktask = new ClockTick(bus);
85              Timer timer = new Timer("clock", true);
86              timer.scheduleAtFixedRate(clocktask, CLOCKDELAY, CLOCKPERIOD);
87          }
88      }
