282         debug("Permanent storage size: $%04X",  read_word((moduleAddr + ModuleConst.m_Mem)));

289         }

308         write_word(DPConst.D_WProcQ, 0);  // Write null to the Wait process queue
309         write_word(DPConst.D_SProcQ, 0);  // Write null to the Sleep process queue

402     /**
403      * F$LOAD - Load module(s) from a file.
404      *
405      * Opens a file specified by the pathlist, reads one or more memory
406      * modules from the file into memory, then closes the file. All modules
407      * loaded are added to the system module directory with a use count of 0,
408      * and the first module read is LINKed. The parameters returned are the same
409      * as the F$LINK call and apply only to the first module loaded.
410      *
411      * In order to be loaded, the file must have the "execute"
412      * permission and contain a module or modules that have a proper module
413      * header. The file will be loaded from the working execution directory
414      * unless a complete pathlist is given.
415      *
416      * If you load a file with two modules, both modules will be added to the
417      * module directory with a use count of 0. Then the first module will be
418      * LINKed giving it a use count of 1. Second time you load the same file,
419      * the first module will get a link count of 2, while the second will still
420      * have link count 0.
421      *  - INPUT:
422      *    - (X) = Address of pathlist (file name)
423      *    - (A) = Language / type (0 = any language / type)
424      *  - OUTPUT:
425      *    - (X) = Advanced past pathlist
426      *    - (Y) = Primary module entry point address
427      *    - (U) = Address of module header
428      *    - (A) - Language / type
429      *    - (B) = Attributes / revision level
430      */
431     public void f_load() {
432         StringBuilder upath = new StringBuilder();
433         byte[] modhead = new byte[14];
434         DevDrvr dev;
435         PathDesc fd;
436         int modname = 0;
437         int moduleSize = 0;
438         int langType = 0;
439 
440         getpath(x, upath, true); // upath now contains the UNIX path to the file.
441         f_prsnam();
442 
443         LOGGER.debug("f_load: {}", upath.toString());
444 
445         dev = find_device(upath.toString());
446         if (dev == null) {
447             sys_error(ErrCodes.E_MNF);
448             LOGGER.debug("f_load: unable to find device");
449             return;
450         }
451         //fd = dev.open(upath.substring(dev.getMntPoint().length()), 5, false);
452         fd = dev.open(upath.toString(), 5, false);
453         if (fd == null) {
454             LOGGER.debug("f_load: unable to open path");
455             sys_error(ErrCodes.E_PNNF);
456             return;
457         }
458 
459         boolean first = true;
460         // Read 1 or more modules into memory
461         while (true) {
462             if (fd.read(modhead, 14) == -1)
463                 break;
464             //debugBuffer(modhead, 14);
465             moduleSize = getWordFromBuffer(modhead, ModuleConst.m_Size);
466             d.set(moduleSize);
467             debug("Allocating module size: $%X #%d", d.intValue(), d.intValue());
468             f_srqmem();            // Request memory of D size - returned in U
469             int moduleAddr = u.intValue();
470             copyBufferToMemory(modhead, moduleAddr, 14);   // copy the header
471             readIntoMemory(fd, moduleAddr + 14, moduleSize - 14); // Read the rest
472             //debugMemory(moduleAddr, moduleSize);
473             add_to_mdir(moduleAddr);
474             if (first) {
475                 first = false;
476                 modname = moduleAddr + getWordFromBuffer(modhead, ModuleConst.m_Name);
477                 langType = getByteFromBuffer(modhead, ModuleConst.m_Type);
478             }
479         }
480         fd.close();
481 
482         x.set(modname);
483         a.set(langType);  // (A) - Language / type
484         f_link();
485     }

489      * process. The module's link count is decremented, and the module is
490      * destroyed and its memory deallocated when the link count equals zero.
491      * The module will not be destroyed if in use by any other process(es)
492      * because its link count will be non-zero.
493      * INPUT:
494      * (U) = Address of module header

496     public void f_unlink() {

498 
499         debug("f_unlink: u=%04X", u.intValue());
500

503                 newcnt = read(mdirp + 2);

507                     d.set(read_word(u.intValue() + ModuleConst.m_Size)); // Module size

509                 }
510                 return;
511             }
512         }
513         sys_error(ErrCodes.E_MNF);

646      * getpath: get the path into a UNIX form, take into account the
647      * execution directory.
648      *
649      * @param mem - location in memory where the path is. Terminated by
650      * high-order bit.
651      * @param unixPath - buffer to store the result in. (side-effect)
652      * @param useXDir - If set, then prefix execution directory on relative paths.
653      * @return value is the end of the path. You usually set register x to that.
654      */
655     private int getpath(Register memReg, StringBuilder unixPath, boolean useXDir) {
656         int mem = memReg.intValue();
657         int mp;
658 
659         /*
660          * When you do a "load filename" in basic09, getpath gets
661          * called with leading spaces in filename
662          */
663         for (mp = mem; read(mp) == ' '; mp++)
664            ;
665 
666         // If the path is absolute, prepend the offset into the UNIX fs
667         if (read(mp) == '/') {
668             unixPath.append("");
669         } else {
670             if (useXDir)
671                 unixPath.append(initialCxd);
672             else
673                 unixPath.append(initialCwd);
674             unixPath.append("/");

736     /**
737      * F$MEM: Used to expand or contract the process' data memory area. The new
738      * size requested is rounded up to the next 256-byte page boundary.
739      * Additional memory is allocated contiguously upward (towards higher
740      * addresses), or deallocated downward from the old highest address. If
741      * D = 0, then the current upper bound and size will be returned.
742      *
743      * This request can never return all of a process' memory, or the
744      * page in which its SP register points to.
745      *
746      * In Level One systems, the request may return an error upon an
747      * expansion request even though adequate free memory exists. This is
748      * because the data area is always made contiguous, and memory requests
749      * by other processes may fragment free memory into smaller, scattered
750      * blocks that are not adjacent to the caller's present data area.
751      *
752      * Level Two systems do not have this restriction because of the
753      * availability of hardware for memory relocation, and because each process
754      * has its own "address space".
755      *  - INPUT:
756      *   - (D) = Desired new memory area size in bytes.
757      *  - OUTPUT:
758      *   - (Y) = Address of new memory area upper bound.
759      *   - (D) = Actual new memory area size in bytes.
760      * FIXME: Must be able to shrink data space as well.
761      */
762     void f_mem() {
763         LOGGER.debug("f_mem: {}", d);
764         Process currProc = getCurrentProcess();
765         int needPages = (d.intValue() + 255) / PAGE_SIZE;
766 
767         if (d.intValue() == 0) {
768             int pages = currProc.getAllocatedPages();
769             int datapage = currProc.getUserAddress();
770             d.set(pages * PAGE_SIZE);
771             y.set((datapage + pages) * PAGE_SIZE);
772         } else {
773             int userAddress = currProc.getUserAddress();
774             int startPage;
775             if (userAddress == 0) { // Nothing allocated currently.
776                 startPage = findPages(currProc, needPages);
777             } else {
778                 startPage = expandPages(currProc, needPages);
779             }
780             if (startPage == 0) {
781                 sys_error(ErrCodes.E_NoRam);
782             }
783         }
784     }

1006      * F$SchBit: Search bit map for a free area.
1007      * This system mode service request searches the specified allocation
1008      * bit map starting at the "beginning bit number" for a free
1009      * block (cleared bits) of the required length.
1010      *
1011      * If no block of the specified size exists, it returns with the
1012      * carry set, beginning bit number and size of the largest block.
1013      *
1014      *  - INPUT:
1015      *   - (X) = Beginning address of a bit map.
1016      *   - (D) = Beginning bit number.
1017      *   - (Y) = Bit count (free bit block size).
1018      *   - (U) = End of bit map address.
1019      *  - OUTPUT:
1020      *   - (D) = Beginning bit number.
1021      *   - (Y) = Bit count.
1022      */
1023     void f_schbit() {
1024         int from, to, bit;
1025         int candidateFrom, candidateLen = 0;
1026         int maxFoundFrom = 0;
1027         int maxFoundLen = 0;
1028 
1029         from = d.intValue();
1030         to = u.intValue();
1031         candidateFrom = from;
1032         while (from < to) {
1033             bit = read(x.intValue() + (from / 8)) & (0x80 >> (from % 8));
1034             if (bit != 0) {
1035                 // bit is taken, we can extend no further
1036                 if (candidateLen < y.intValue()) {
1037                     candidateLen = 0; // Reset candidateLen to 0
1038                     candidateFrom = from + 1; // The next start can be earliest from next bit.
1039                 }
1040             } else {
1041                 // bit is free
1042                 candidateLen++;
1043                 if (candidateLen > maxFoundLen) {
1044                     maxFoundFrom = from - candidateLen + 1;
1045                     maxFoundLen = candidateLen;
1046                 }
1047             }
1048             if (candidateLen == y.intValue()) {
1049                 break;
1050             }
1051             from++;
1052         }
1053 
1054         if (candidateLen < y.intValue()) {
1055             sys_error(ErrCodes.E_NoRam);
1056             d.set(maxFoundFrom);
1057             y.set(maxFoundLen);
1058         } else {
1059             d.set(candidateFrom);
1060             y.set(candidateLen);
1061         }
