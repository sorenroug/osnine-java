14  /**
15   * A service to handle telnet connections.
16   * It listens on port 2323. When a client connects, then it
17   * launches two threads. One for reading and one for writing.
18   * These threads then call the ACIA.
19   */
20  public class AciaTelnetUI implements Runnable {
21  
22      private static final Logger LOGGER = 
23              LoggerFactory.getLogger(AciaTelnetUI.class);

34          false,   // Transmit binary
35          false,   // Echo
36          false,   // Reconnection
37          false,   // Suppress GA
38          false,   // Approx message size
39          false,   // Status
40          false,   // Timing mark

58      public void run() {
59          final int portNumber = 2323;
60          ServerSocket serverSocket;
61  
62          LOGGER.debug("AciaTelnetUI thread started");
63          try {
64              serverSocket = new ServerSocket(portNumber);
65          } catch (Exception e) {
66              LOGGER.error("Unable to create listening socket");
67              return;
68          }
69          synchronized(this) {
70              activeSocket = null;
71          }
72  
73          try {
74              while (true) {
75                  LOGGER.info("Waiting for connection");
76                  Socket socket = serverSocket.accept();
77                  synchronized(this) {
78                      if (activeSocket != null) {
79                          OutputStreamWriter w = new OutputStreamWriter(socket.getOutputStream());
80                          w.write("\r\nAnother session is active\r\n\r\n");
81                          w.flush();
82                          socket.close();
83                          continue;
84                      }
85                      socket.setKeepAlive(true);
86                      clientOut = socket.getOutputStream();
87                      clientIn = socket.getInputStream();
88                      activeSocket = socket;
89                  }
90                  acia.setDCD(true);
91                  resetModes();
92  
93                  reader = new Thread(new LineReader(this), "acia-in");
94                  reader.start();
95  
96                  term = GO51Terminal.NORMAL;
97                  writer = new Thread(new LineWriter(), "acia-out");
98                  writer.start();
99                  synchronized(this) {
100                     listeners = 2;
101                 }
102             }
103         } catch (IOException e) {
104             LOGGER.error("IOException", e);
105             return;
106         }
107     }

237     /**
238      * Thread to write to socket.
239      * If it gets a write exception, then it shall tell the reader
240      * to stop and it shall end the thread.
241      * If it gets an InterruptedException then it shall just stop.
242      */
243     class LineWriter implements Runnable {
244 
245         /**
246          * Wait for a value from the Acia to transmit to the client terminal.
247          * Send it, if exception, assume the connection is broken and exit.
248          */
249         public void run() {
250             LOGGER.debug("Writer thread started");
251             try {
252                 willDo(TelnetState.SUPPRESS_GA);
253                 willDo(TelnetState.ECHO);
254                 while (true) {
255                     int val = waitForValueToTransmit();
256                     sendTelnetClient(val);
257                     if (Thread.interrupted())
258                         throw new InterruptedException();
259                 }
260             } catch (InterruptedException e) {
261                 LOGGER.error("InterruptException");
262                 endSession();
263             } catch (Exception e) {
264                 LOGGER.error("Broken connection", e);
265                 interruptReader();
266             }
267         }
268     }

272 /**
273  * Thread to listen to incoming data from a telnet client.
274  * If it gets a read exception, then it shall tell the writer
275  * to stop and it shall end the thread.
276  * If it gets an InterruptedException then it shall just stop.
277  */
278 class LineReader implements Runnable {
279 
280     private static final Logger LOGGER = 
281             LoggerFactory.getLogger(LineReader.class);
282 
283     private TelnetState state = TelnetState.NORMAL;
284 
285     private AciaTelnetUI handler;
286 
287     /** Characters read from the telnet client. */
288     private volatile int charsRead;
289 
290     /**
291      * Constructor.
292      */
293     LineReader(AciaTelnetUI handler) {
294         this.handler = handler;
295     }
296 
297     /**
298      * Since this waits in read() we can busy loop on the rest.
299      */
300     public void run() {
301         LOGGER.debug("Reader thread started");
302         Timer timer = new Timer("escape", true);
303         try {
304             while (true) {
305                 int receiveData = handler.read();
306                 if (Thread.interrupted())
307                     throw new InterruptedException();
308                 LOGGER.debug("Received {}", receiveData);
309                 if (receiveData == -1) {
310                     LOGGER.info("Reader lost connection");
311                     handler.interruptWriter();
312                     break;
313                 }
314                 charsRead++;
315                 if (receiveData == 27) {
316                     TimeoutEscape timeout = new TimeoutEscape(this, charsRead);
317                     timer.schedule(timeout, 100);
318                 }
319                 state = state.handleCharacter(receiveData, handler);
320             }
321         } catch (InterruptedException e) {
322             LOGGER.error("InterruptException", e);
323             handler.endSession();
324         } catch (Exception e) {
325             LOGGER.error("Exception", e);
326             handler.interruptWriter();
327         }
328     }
329 
330     void resetState(int charsReadAtStart) {
331         try {
332             if (charsRead == charsReadAtStart) {
333                 state = state.handleCharacter(0, handler);
334             }
335         } catch (IOException e) {
336           // Do nothing
337         }
338     }
339 }
