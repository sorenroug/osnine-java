17       * Load a short program into memory.

19      private void loadProg(int[] instructions) {
20          writeword(0xfffe, LOCATION);
21          int respc = myTestCPU.read_word(0xfffe);
22          assertEquals(LOCATION, respc);
23  
24          for (int i = 0; i < instructions.length; i++) {
25              myTestCPU.write(i + LOCATION, instructions[i]);
26          }
27          myTestCPU.reset();
28      }

434     @Test
435     public void testJSR() {
436         // Set up a word to test at address 0x205
437         writeword(0x205, 0x03ff);
438         // Set register D
439         myTestCPU.d.set(0x105);
440         // Set register Y to point to that location minus 5
441         setY(0x200);
442         // Set register S to point to 0x915
443         myTestCPU.s.set(0x915);
444         // Two bytes of instruction
445         myTestCPU.write(0xB00, 0xAD);
446         myTestCPU.write(0xB01, 0xAB);
447         myTestCPU.write(0xB02, 0x11); // Junk
448         myTestCPU.write(0xB03, 0x22); // Junk
449         setPC(0xB00);
450         myTestCPU.cc.clear();
451         myTestCPU.execute();
452         chkCC_A_B_DP_X_Y_S_U(0, 0x01, 0x05, 0, 0, 0x200, 0x913, 0);
453         assertEquals(0x200, myTestCPU.y.intValue());
454         assertEquals(0x105, myTestCPU.d.intValue());
455         assertEquals(0x913, myTestCPU.s.intValue());
456         assertPC(0x305);
